-- SQL schema para uso no DBeaver (compatível com SQLite/Postgres)
PRAGMA foreign_keys = ON;

CREATE TABLE admins (
  id INTEGER PRIMARY KEY,
  nome TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  senha TEXT NOT NULL
);

CREATE TABLE psicologos (
  id INTEGER PRIMARY KEY,
  nome TEXT NOT NULL,
  titulo TEXT NOT NULL,
  bio TEXT NOT NULL,
  foto TEXT,
  ativo INTEGER DEFAULT 1
);

CREATE TABLE servicos (
  id INTEGER PRIMARY KEY,
  nome TEXT NOT NULL,
  descricao TEXT,
  valor NUMERIC(10,2) NOT NULL,
  duracao INTEGER NOT NULL,
  modalidade TEXT NOT NULL,
  ativo INTEGER DEFAULT 1
);

CREATE TABLE pacientes (
  id INTEGER PRIMARY KEY,
  nome TEXT NOT NULL,
  email TEXT NOT NULL,
  telefone TEXT NOT NULL,
  criado_em TEXT DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE horarios_disponiveis (
  id INTEGER PRIMARY KEY,
  data TEXT NOT NULL,
  hora TEXT NOT NULL,
  disponivel INTEGER DEFAULT 1
);

CREATE TABLE agendamentos (
  id INTEGER PRIMARY KEY,
  paciente_id INTEGER NOT NULL,
  psicologo_id INTEGER,
  servico_id INTEGER NOT NULL,
  data TEXT NOT NULL,
  hora TEXT NOT NULL,
  mensagem TEXT,
  status TEXT DEFAULT 'Pendente',
  criado_em TEXT DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (paciente_id) REFERENCES pacientes(id) ON DELETE CASCADE,
  FOREIGN KEY (psicologo_id) REFERENCES psicologos(id) ON DELETE SET NULL,
  FOREIGN KEY (servico_id) REFERENCES servicos(id) ON DELETE CASCADE
);

-- Observações:
-- - As tabelas refletem diretamente os dados usados em index.html, agendar.html e admin.html.
-- - Horários disponíveis são usados pelo calendário e pelo painel administrativo.
-- - Serviços, pacientes e agendamentos são normalizados para manter os relacionamentos.
